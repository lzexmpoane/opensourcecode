long a[101];

void sswap(long& sa,long& sb)

{

    long sc=sa;

    sa=sb;

    sb=sc;

}

void qsort(long l,long r)

{

    int i=l,j=r,mid=a[(l+r)/2];

    while(i<=j)

    {

        while(a[i]<mid)

            i++;

        while(a[j]>mid)

            j--;

        if(i<=j)

        {

            sswap(a[i],a[j]);

            i++;j--;

        }

    }

    if(l<j)qsort(l,j);

    if(i<r)qsort(i,r);

}
