#include <iostream>

#include <cstdio>

using namespace std;

typedef long long ll;

ll c[205][2],dp[10000];

ll n,m;

int main()

{

    cin>>m>>n;

    for(int i=1;i<=n;i++)

    {

        cin>>c[i][0]>>c[i][1];

    }

    for(ll i=1;i<=n;i++)

    {

        for(ll j=c[i][0];j<=m;j++)

        {

            dp[j]=max(dp[j],dp[j-c[i][0]]+c[i][1]);

        }

    }

    cout<<"max="<<dp[m];

}
