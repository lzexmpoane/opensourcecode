</head>\n\

<p><a href=\"#\" onClick=\"javascript :history.back(-1);\">返回</a><p>\n\

<font face=\"Courier New\">";

string str="";

int main()

{

    freopen("code.cpp","r",stdin);

    string s;

    do

    {

        getline(cin,s);

        if(s=="$")break;

        str+=(s+'\n');

    }while(s!="$");

    system("cls");

    cout<<head;

    cout<<"<p>";

    for(int i=0;i<str.length();i++)

    {

        if(str[i]=='\"')cout<<"&quot;";

        else if(str[i]=='<')cout<<"&lt;";

        else if(str[i]=='&')cout<<"&amp;";

        else if(str[i]=='>')cout<<"&gt;";

        else if(str[i]==' ')cout<<"&nbsp;";

        else if(str[i]=='\n')cout<<"</p><p>";

        else if(str[i]=='\t')cout<<"&nbsp;&nbsp;&nbsp;&nbsp;";

        else cout<<str[i];

    }

    cout<<"</p></font>";

    return 0;

}
